            // Undefined is the type of a variable before it is set.
            var bananas;
            
            // The type will be set to undefined
            console.log(typeof bananas);
            
            // Set the variable to a string
            bananas = "Bananas";
            
            // Check the type again; it should be a string now.
            console.log(typeof bananas);

null

            
            // Null means a variable is defined, but has no value.
            var blink182 = null;
            
            // The type will be set to undefined
            console.log(typeof blink182);
            
            // Null isn't the same as any other value.
            console.log(null == true);
            console.log(null == false);
            console.log(null == 0);
            console.log(null == "");

Arrays

            
            // Arrays are ordered collections of any value.
            var images = ["puppy.png", "duck.jpg", "portal.gif"];
            
            // Elements in an array are accessed numerically, starting from 0.
            console.log(images[0]); // puppy.png
            console.log(images[1]); // duck.jpg
            console.log(images[2]); // portal.gif
            
            // You can add a new value to the end of an array with push()
            images.push("northrend.tga");
            console.log(images[3]); // northrend.tga
            
            // And delete the last value using pop()
            images.pop();
            console.log(images[3]); // undefined
            
            // You can change the value of any array location
            images[0] = "kitty.png";
            console.log(images[0]); // kitty.png

Getting data from the user

There exist three ways of getting data from the user using pure JavaScript.

  • console.log(message): Shows user the message; waits until user clicks "OK".
  • confirm(message): Shows user the message; waits until they click "OK" or "Cancel".
  • prompt(message, placeholder): Shows user the message, allows them to write in a single-line text box that is pre-filled with placeholder, and waits until they click "OK" or "Cancel".

Flow Control: if( )...else

Perform different actions if a condition evaluates to true or false.

            var animal = "leopard";
            if (animal === "leopard") {
                console.log("The animal is a Leopard!");
            } else {
                console.log("The animal is not a Leopard!");
            }

Flow Control: else if( )

if()...else can be chained with else if()

var animal = "leopard";
if (animal === "leopard") {
    console.log("The animal is a Leopard!");
} else if (animal === "tiger") {
    console.log("The animal is a Tiger!");
} else if (animal === "panther") {
    console.log("The animal is a Panther!");
} else {
    console.log("The animal is something I don't know about!");
}

Logical Operator: && and

&& allows a statement to be true only if both statements are true.

            var animal = "leopard", wild = true;
            if (animal === "leopard" && wild === true) {
            	// This block will be executed only if animal is leopard and wild is true
            	console.log("The animal is a Wild Leopard!");
            }

Logical Operator: || or

|| allows a statement to be true if either statement is true.

            var animal = "leopard";
            if (animal === "leopard" || animal === "tiger") {
            	// This block will be executed if animal is either leopard or tiger
            	console.log("The animal is a Leopard or Tiger!");
            }

Logical Operator: ! not

! flips true to false & false to true.

            var isAdmin = false;
            if (!isAdmin) {
            	// This block will be executed only if isAdmin is set to false
            	console.log("You are not an Admin!");
            }

Switching Types

            // When using prompt(), user-provided data is alway given as a string.
            var userNumber = prompt("Enter any number.");
            
            // If the user enters "1", it won't work as expected with other numbers.
            console.log("1" + 2) // 12
            
            // To fix this, you can use parseFloat()
            var numberAsString = "1";
            var numberAsNumber = parseFloat(numberAsString);
            
            console.log(numberAsNumber + 2); // 3
            
            // To specifically get an integer, you can use parseInt()
            var intAsString = "5.201949583939";
            var intAsNumber = parseInt(intAsString);
            
            console.log(intAsNumber); // 5
